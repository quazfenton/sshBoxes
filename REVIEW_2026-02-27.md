#### 2. Web Terminal (CRITICAL - Must Have)

**Why:** CLI-only approach limits adoption to power users. Web terminal opens access to 90%+ of users who don't use SSH regularly.

**Recommended Stack:**
- **Frontend:** Xterm.js + React
- **Backend:** WebSocket → SSH bridge (gotty or wetty)
- **Protocol:** WebSocket with SSH over it

**Implementation:**
```typescript
// web/websocket-bridge.ts
import { WebSocketServer } from 'ws';
import { SSHConnection } from 'node-ssh';

const wss = new WebSocketServer({ port: 8080 });

wss.on('connection', (ws, req) => {
  const token = extractToken(req);
  const box = await provisionBox(token);
  
  const ssh = new SSHConnection();
  await ssh.connect({
    host: box.host,
    port: box.port,
    username: 'guest',
    privateKey: box.privateKey
  });
  
  // Bridge WebSocket to SSH
  ws.on('message', (data) => ssh.stdin(data));
  ssh.stdout.on('data', (data) => ws.send(data));
});
```

**Timeline:** 1 week
**Impact:** 10x adoption increase

// mcp/server.ts
import { McpServer } from '@modelcontextprotocol/sdk';

const server = new McpServer({
  name: 'sshboxes',
  version: '1.0.0',
});

server.tool('provision-ephemeral-box', {
  description: 'Create an ephemeral SSH box for code execution',
  inputSchema: {
    type: 'object',
    properties: {
      profile: { type: 'string', enum: ['dev', 'debug', 'secure-shell'] },
      ttl: { type: 'number', default: 1800 }
    }
  }
}, async ({ profile, ttl }) => {
  const box = await provisionBox(profile, ttl);
  return { 
    host: box.host, 
    port: box.port,
    connection: `ssh -p ${box.port} guest@${box.host}`
  };
});
